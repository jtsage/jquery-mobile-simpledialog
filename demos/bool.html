jQuery
$(document).delegate('#simplebool', 'click', function() {
  $(this).simpledialog({
    'mode' : 'bool',
    'prompt' : 'How about it?',
    'useModal': true,
    'buttons' : {
      'OK': {
        click: function () {
          $('#dialogoutput').text('OK');
        }
      },
      'Cancel': {
        click: function () {
          $('#dialogoutput').text('Cancel');
        },
        icon: "delete",
        theme: "c"
      }
    }
  })
})
